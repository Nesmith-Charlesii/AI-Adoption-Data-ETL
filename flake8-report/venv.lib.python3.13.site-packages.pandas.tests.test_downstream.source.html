191 def test_missing_required_dependency():
192 # GH 23868
193 # To ensure proper isolation, we pass these flags
194 # -S : disable site-packages
195 # -s : disable user site-packages
196 # -E : disable PYTHON* env vars, especially PYTHONPATH
197 # https://github.com/MacPython/pandas-wheels/pull/50
198  
199 pyexe = sys.executable.replace("\\", "/")
200  
201 # We skip this test if pandas is installed as a site package. We first
202 # import the package normally and check the path to the module before
203 # executing the test which imports pandas with site packages disabled.
204 call = [pyexe, "-c", "import pandas;print(pandas.__file__)"]
205 output = subprocess.check_output(call).decode()
206 if "site-packages" in output:
207 pytest.skip("pandas installed as site package")
208  
209 # This test will fail if pandas is installed as a site package. The flags
210 # prevent pandas being imported and the test will report Failed: DID NOT
211 # RAISE <class 'subprocess.CalledProcessError'>
212 call = [pyexe, "-sSE", "-c", "import pandas"]
213  
214 msg = (
215 rf"Command '\['{pyexe}', '-sSE', '-c', 'import pandas'\]' "
216 "returned non-zero exit status 1."
217 )
218  
219 with pytest.raises(subprocess.CalledProcessError, match=msg) as exc:
220 subprocess.check_output(call, stderr=subprocess.STDOUT)

305 @pytest.mark.parametrize("dtype", ["M8[ns]", "m8[ns]"])
306 def test_from_obscure_array(dtype, array_likes):
307 # GH#24539 recognize e.g xarray, dask, ...
308 # Note: we dont do this for PeriodArray bc _from_sequence won't accept
309 # an array of integers
310 # TODO: could check with arraylike of Period objects
311 arr, data = array_likes
312  
313 cls = {"M8[ns]": DatetimeArray, "m8[ns]": TimedeltaArray}[dtype]
314  
315 depr_msg = f"{cls.__name__}.__init__ is deprecated"
316 with tm.assert_produces_warning(FutureWarning, match=depr_msg):
317 expected = cls(arr)
318 result = cls._from_sequence(data, dtype=dtype)
319 tm.assert_extension_array_equal(result, expected)
320  
321 if not isinstance(data, memoryview):
322 # FIXME(GH#44431) these raise on memoryview and attempted fix
323 # fails on py3.10
324 func = {"M8[ns]": pd.to_datetime, "m8[ns]": pd.to_timedelta}[dtype]
325 result = func(arr).array
326 expected = func(data).array
327 tm.assert_equal(result, expected)
