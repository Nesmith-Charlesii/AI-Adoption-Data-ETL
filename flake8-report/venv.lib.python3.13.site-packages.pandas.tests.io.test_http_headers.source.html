127 def test_request_headers(responder, read_method, httpserver, storage_options):
128 expected = pd.DataFrame({"a": ["b"]})
129 default_headers = ["Accept-Encoding", "Host", "Connection", "User-Agent"]
130 if "gz" in responder.__name__:
131 extra = {"Content-Encoding": "gzip"}
132 if storage_options is None:
133 storage_options = extra
134 else:
135 storage_options |= extra
136 else:
137 extra = None
138 expected_headers = set(default_headers).union(
139 storage_options.keys() if storage_options else []
140 )
141 httpserver.serve_content(content=responder(expected), headers=extra)
142 result = read_method(httpserver.url, storage_options=storage_options)
143 tm.assert_frame_equal(result, expected)
144  
145 request_headers = dict(httpserver.requests[0].headers)
146 for header in expected_headers:
147 exp = request_headers.pop(header)
148 if storage_options and header in storage_options:
149 assert exp == storage_options[header]
150 # No extra headers added
151 assert not request_headers
